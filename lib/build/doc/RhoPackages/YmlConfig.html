def initialize(config_yml_file)
  @config = YAML::load_file(config_yml_file)

  @platfrom_suffix = case RUBY_PLATFORM
  when /(win|w)32$/
    'win'
  when /darwin/
    'mac'
  else
    fail
  end
end

def package_file(package_name)
  file = package_name
  file += ".#{@platfrom_suffix}" if platform_specific?(package_name)
  file
end
